if (result.entities.length > 0) {
    // 1) Grab the right GUID
    const record    = result.entities[0];
    const recordId  = record.nf1pbm_maturityinstructionsid;

    // 2) Build your View action
    const viewAction = {
      actionLabel: "View",
      eventHandler: function () {
        const options = {
          entityName: CONSTANTS.ENTITY_NAMES.MATURITY_INSTRUCTIONS,
          entityId:   recordId
        };
        Xrm.Navigation.openForm(options)
          .then(function () {
            console.log("Form opened for ID:", recordId);
          })
          .catch(function (err) {
            console.error("openForm failed:", err);
          });
      }
    };

    // 3) Show the global notification
    Xrm.App.addGlobalNotification({
      type:            2,              // app-level banner :contentReference[oaicite:0]{index=0}
      level:           2,              // error severity :contentReference[oaicite:1]{index=1}
      message:         formatString(CONSTANTS.MESSAGES.DUPLICATE_RECORD, recordId),
      showCloseButton: true,
      action:          viewAction
    }).then(
      function (notifId) { console.log("Notification ID:", notifId); },
      function (err)     { console.error("Notification error:", err); }
    );

    makeFormReadOnly(executionContext);
    return []; // stop further processing
}
