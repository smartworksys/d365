action: {
  actionLabel: `View ${recordTitle}`,
  eventHandler: function () {
    // 1) suppress unsaved-changes dialog
    formContext.data.entity.attributes.get().forEach(attr => attr.setSubmitMode("never"));

    // 2) clear the banner
    Xrm.App.clearGlobalNotification(duplicateNotifId)
      .then(() => console.log("Cleared duplicate warning"))
      .catch(err => console.error(err));

    // 3) hide Save buttons via your ribbon-flag
    formContext.getAttribute("new_hideSaveButtons").setValue(true);
    formContext.getAttribute("new_hideSaveButtons").fireOnChange();
    formContext.ui.refreshRibbon();

    // 4) navigate to the existing record
    Xrm.Navigation.openForm({
      entityName: CONSTANTS.ENTITY_NAMES.MATURITY_INSTRUCTIONS,
      entityId:   recordGuid
    }).then(function () {
      window.location.reload();
    }).catch(function (err) {
      console.error("openForm error", err);
    });
  }
}
